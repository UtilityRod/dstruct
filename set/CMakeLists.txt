if(SET)
    get_property(added_dstructs GLOBAL PROPERTY ADDED_DSTRUCTS)

    if (NOT added_dstructs MATCHES "SET")
        set_property(GLOBAL PROPERTY ADDED_DSTRUCTS "${added_dstructs} SET")
        message(STATUS "[+] adding Set to dstruct")
        target_sources(dstruct PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/set.c)
        target_include_directories(dstruct PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
    endif()
endif()
