if(TREE)
    get_property(added_dstructs GLOBAL PROPERTY ADDED_DSTRUCTS)

    if (NOT added_dstructs MATCHES "TREE")
        set_property(GLOBAL PROPERTY ADDED_DSTRUCTS "${added_dstructs} TREE")

        target_sources(dstruct PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/tree.c)
        target_include_directories(dstruct PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
    endif()
endif()
